<script>
document.addEventListener('DOMContentLoaded', function () {
  const root = document.getElementById('pricing');
  if (!root) return;

  const buttons = root.querySelectorAll('.price-toggle-btn');

  function parseMoney(str){
    if(!str) return NaN;
    const m = String(str).replace(/,/g,'').match(/[\d.]+/);
    return m ? parseFloat(m[0]) : NaN;
  }

  function isHourly(html){
    return /\/\s*hr/i.test(html);
  }

  function withUSD(html){
    // If hourly, do not add USD
    if (isHourly(html)) return html;

    // If USD already present, keep as is
    if (/USD\b/i.test(html)) return html;

    // Most dataset values are plain like "$230" or "$975"
    // Insert USD after the first price token
    // If there are spans afterwards, we still append immediately after the number.
    const m = html.match(/^\s*[^0-9$-]*[\$£€]?\s*[\d.,]+/);
    if (!m) return html + '<span class="usd"> USD</span>';

    const idx = m[0].length;
    return html.slice(0, idx) + '<span class="usd"> USD</span>' + html.slice(idx);
  }

  function setMode(size){
    buttons.forEach(b => b.classList.toggle('active', b.dataset.size === size));

    root.querySelectorAll('.price-text').forEach(priceEl => {
      const ds = priceEl.dataset;

      const perEl = priceEl.closest('.wp-block-column, .price-card-container')?.querySelector('.price-per-person-text')
                  || priceEl.parentElement?.querySelector('.price-per-person-text');

      const saveEl = priceEl.closest('.wp-block-column, .price-card-container')?.querySelector('.price-save-text')
                   || priceEl.parentElement?.querySelector('.price-save-text');

      const priceKey = (size === 'solo') ? 'priceSolo' : ('price' + size);
      const soloStr  = ds.priceSolo || '';
      const currStr  = ds[priceKey] || '';

      // Update price HTML and add USD for non-hourly
      if (currStr){
        priceEl.innerHTML = withUSD(currStr);
      }

      // per-learner toggle
      if (perEl) perEl.classList.toggle('hidden', size === 'solo');

      // Save text
      if (saveEl){
        if (size === 'solo'){
          saveEl.innerHTML = (saveEl.dataset.saveSolo || '').trim();
        } else {
          const soloNum = parseMoney(soloStr);
          const currNum = parseMoney(currStr);
          const diff = (isFinite(soloNum) && isFinite(currNum)) ? Math.round(soloNum - currNum) : 0;
          saveEl.innerHTML = diff > 0 ? `Save $${diff} each` : '';
        }
      }
    });
  }

  buttons.forEach(b => b.addEventListener('click', () => setMode(b.dataset.size)));
  setMode('solo');
});
</script>

<script>
document.addEventListener('DOMContentLoaded', function () {
    const questions = document.querySelectorAll('.faq-question');
    questions.forEach(question => {
        question.addEventListener('click', () => {
            const item = question.parentElement;
            const answer = item.querySelector('.faq-answer');
            const caret = item.querySelector('.caret');
            const isOpen = item.classList.contains('active');

            if (isOpen) {
                answer.style.maxHeight = null;
                item.classList.remove('active');
                caret.classList.remove('open');
            } else {
                answer.style.maxHeight = answer.scrollHeight + 'px';
                item.classList.add('active');
                caret.classList.add('open');
            }
        });
    });
});
</script>

(function(){
  // 1) Stable visitor id
  function uuidv4(){
    return ([1e7]+-1e3+-4e3+-8e3+-1e11).replace(/[018]/g,c=>
      (c ^ crypto.getRandomValues(new Uint8Array(1))[0] & 15 >> c / 4).toString(16)
    );
  }
  function getCookie(name){
    return document.cookie.split('; ').find(r=>r.startsWith(name+'='))?.split('=')[1];
  }
  function setCookie(name,value,days){
    const d = new Date(); d.setTime(d.getTime()+days*864e5);
    document.cookie = `${name}=${value}; expires=${d.toUTCString()}; path=/; SameSite=Lax`;
  }

  // lp_view_id cookie
  let viewId = getCookie('lp_view_id');
  if(!viewId){ viewId = uuidv4(); setCookie('lp_view_id', viewId, 400); }
  window.lpViewId = viewId;

  // 2) Assign a form variant if not decided yet (50/50)
  let variant = getCookie('lp_variant');
  if(!variant){
    variant = Math.random() < 0.5 ? 'A' : 'B';
    setCookie('lp_variant', variant, 90);
  }
  window.lpVariant = variant;

  // 3) Optional: expose UTM params
  const p = new URLSearchParams(location.search);
  window.lpUtm = {

    content: p.get('utm_content') || '',
    term: p.get('utm_term') || ''
  };

  // 4) (Nice-to-have) fire a GTM dataLayer event for a “form_view”
