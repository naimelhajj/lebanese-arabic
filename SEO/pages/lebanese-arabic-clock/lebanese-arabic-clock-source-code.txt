<script>
(function () {
  let initialized = false;

  function init() {
    const wrapper = document.getElementById('lebanese-clock-app-wrapper');
    const clockApp = document.getElementById('lebanese-clock-app');
    const spinner = document.getElementById('loading-spinner');

    if (initialized || !wrapper || !clockApp || !spinner) return;
    initialized = true;

    // Define elements inside init to ensure they exist when called.
    const hourHand = clockApp.querySelector('#hour-hand');
    const minuteHand = clockApp.querySelector('#minute-hand');
    const secondHand = clockApp.querySelector('#second-hand');
    const digitalTimeEl = clockApp.querySelector('#digital-time');
    const lebaneseTimeArabicEl = clockApp.querySelector('#lebanese-time-arabic');
    const lebaneseTimeTranslitEl = clockApp.querySelector('#lebanese-time-translit');
    const timeModeBtn = clockApp.querySelector('#time-mode-btn');
    const infoText = clockApp.querySelector('#info-text');
    const shareBtn = clockApp.querySelector('#share-btn');
    const embedBtn = clockApp.querySelector('#embed-btn');
    const copyFeedback = clockApp.querySelector('#copy-feedback');
    const clockContainer = clockApp.querySelector('#clock-container');
    
    let clockInterval = null, isRealTime = true, isDragging = false,
        initialHour = 0, initialMinute = 0, lastAngle = 0, cumulativeMinuteDelta = 0;
    
    const numbers = [ { num: 1,  ar: 'وحدة',  translit: 'weHde'   }, { num: 2,  ar: 'تنتين', translit: 'tinten'  }, { num: 3,  ar: 'تلاتة', translit: 'tleteh'  }, { num: 4,  ar: 'أربعة', translit: 'arb3a'   }, { num: 5,  ar: 'خمسة',  translit: 'khamseh' }, { num: 6,  ar: 'ستة',   translit: 'sitteh'  }, { num: 7,  ar: 'سبعة',  translit: 'sab3a'   }, { num: 8,  ar: 'تمانة', translit: 'tmeneh'  }, { num: 9,  ar: 'تسعة',  translit: 'tis3a'   }, { num: 10, ar: 'عشرة',  translit: '3ashra'  }, { num: 11, ar: 'حدعش',  translit: 'Hda3sh'  }, { num: 12, ar: 'تنعش',  translit: 'tna3sh'  } ];
    const minutesText = { 5:{ar:'وخمسة',translit:'w khamseh'},10:{ar:'وعشرة',translit:'w 3ashra'}, 15:{ar:'وربع',translit:'w rebe3'}, 20:{ar:'وتلت',translit:'w tilit'}, 25:{ar:'ونص إلا خمسة',translit:'w noss illa khamseh'},30:{ar:'ونص',translit:'w noss'}, 35:{ar:'ونص وخمسة',translit:'w noss w khamseh'},40:{ar:'إلا تلت',translit:'illa tilit'}, 45:{ar:'إلا ربع',translit:'illa rebe3'},50:{ar:'إلا عشرة',translit:'illa 3ashra'}, 55:{ar:'إلا خمسة',translit:'illa khamseh'} };

    function drawClockMarkers() {
        const markersEl = clockApp.querySelector('#markers'), numbersEl = clockApp.querySelector('#numbers');
        if (!markersEl || !numbersEl) return;
        markersEl.innerHTML = ''; numbersEl.innerHTML = '';
        for (let i = 1; i <= 60; i++) {
            const angle = (i * 6) * Math.PI / 180, isHour = i % 5 === 0, r1 = isHour ? 85 : 90, r2 = 95;
            const x1 = 100 + r1 * Math.sin(angle), y1 = 100 - r1 * Math.cos(angle);
            const x2 = 100 + r2 * Math.sin(angle), y2 = 100 - r2 * Math.cos(angle);
            const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');
            line.setAttribute('x1', x1); line.setAttribute('y1', y1); line.setAttribute('x2', x2); line.setAttribute('y2', y2);
            line.setAttribute('stroke', isHour ? '#1f2937' : '#9ca3af'); line.setAttribute('stroke-width', isHour ? 2 : 1);
            markersEl.appendChild(line);
            if (isHour) {
                const hour = i/5, numR = 72, numX = 100 + numR * Math.sin(angle), numY = 100 - numR * Math.cos(angle);
                const text = document.createElementNS('http://www.w3.org/2000/svg', 'text');
                text.setAttribute('x', numX); text.setAttribute('y', numY + 1); text.classList.add('number'); text.textContent = hour;
                numbersEl.appendChild(text);
            }
        }
    }

    function convertToLebanese(hour, minute) {
        hour = hour % 12; if (hour === 0) hour = 12;
        const hourInfo = numbers.find(n => n.num === hour);
        if (!hourInfo) return { ar: '', translit: ''};
        if (minute === 0) return { ar: `${hourInfo.ar}`, translit: `${hourInfo.translit}` };
        const rounded = Math.round(minute / 5) * 5;
        if (!minutesText[rounded]) {
             const nextHourRaw = (hour === 12) ? 1 : hour + 1;
             const nextHourInfo = (rounded === 0) ? hourInfo : numbers.find(n => n.num === nextHourRaw);
             if (!nextHourInfo) return { ar: '', translit: ''};
             return { ar: `${nextHourInfo.ar}`, translit: `${nextHourInfo.translit}` };
        }
        if (rounded === 35) { const m = minutesText[rounded]; return { ar:`${hourInfo.ar} ${m.ar}`, translit:`${hourInfo.translit} ${m.translit}` }; }
        if (rounded > 30) {
            const nextHourRaw = (hour === 12) ? 1 : hour + 1;
            const nextHourInfo = numbers.find(n => n.num === nextHourRaw);
            if (!nextHourInfo) return { ar: '', translit: ''};
            const m = minutesText[rounded]; return { ar:`${nextHourInfo.ar} ${m.ar}`, translit:`${nextHourInfo.translit} ${m.translit}` };
        } else {
            const m = minutesText[rounded]; return { ar:`${hourInfo.ar} ${m.ar}`, translit:`${hourInfo.translit} ${m.translit}` };
        }
    }

    function setTime(date) {
        const s = date.getSeconds(), m = date.getMinutes(), h = date.getHours();
        const secDeg = (s / 60) * 360, minDeg = (m / 60) * 360 + (s / 60) * 6, hDeg  = (h / 12) * 360 + (m / 60) * 30;
        secondHand.style.transform = `rotate(${secDeg}deg)`; minuteHand.style.transform = `rotate(${minDeg}deg)`; hourHand.style.transform = `rotate(${hDeg}deg)`;
        digitalTimeEl.textContent = `${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
        const phr = convertToLebanese(h, m); lebaneseTimeArabicEl.textContent = phr.ar; lebaneseTimeTranslitEl.textContent = phr.translit;
    }

    function setManualTime(hour, minute) {
        const m = Math.max(0, Math.min(59, Math.floor(minute))), h = ((Math.floor(hour) % 24) + 24) % 24;
        const minDeg = (m / 60) * 360, hDeg = (h / 12) * 360 + (m / 60) * 30;
        secondHand.style.display = 'none'; minuteHand.style.transform = `rotate(${minDeg}deg)`; hourHand.style.transform = `rotate(${hDeg}deg)`;
        digitalTimeEl.textContent = `${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:--`;
        const phr = convertToLebanese(h, m); lebaneseTimeArabicEl.textContent = phr.ar; lebaneseTimeTranslitEl.textContent = phr.translit;
    }

    function startClock() {
        if (clockInterval) clearInterval(clockInterval);
        isRealTime = true; secondHand.style.display = 'block';
        timeModeBtn.textContent = 'Drag the hands to set time'; infoText.textContent = '';
        setTime(new Date()); clockInterval = setInterval(() => setTime(new Date()), 1000);
    }

    function stopClock() {
        if (clockInterval) { clearInterval(clockInterval); clockInterval = null; }
        isRealTime = false; timeModeBtn.textContent = 'Show Current Time'; infoText.textContent = 'You are in interactive mode.';
    }

    function handleInteraction(e) {
        if (!isDragging) return; e.preventDefault();
        const rect = clockContainer.getBoundingClientRect(), cx = rect.left + rect.width / 2, cy = rect.top + rect.height / 2;
        const clientX = e.touches ? e.touches[0].clientX : e.clientX, clientY = e.touches ? e.touches[0].clientY : e.clientY;
        const angRad = Math.atan2(clientY - cy, clientX - cx), current = (angRad * 180 / Math.PI + 90 + 360) % 360;
        let delta = current - lastAngle; if (delta > 180) delta -= 360; if (delta < -180) delta += 360;
        lastAngle = current; cumulativeMinuteDelta += (delta / 6);
        const startTotal = initialHour * 60 + initialMinute, newTotal = Math.round(startTotal + cumulativeMinuteDelta);
        const newHour = Math.floor(newTotal / 60), newMinute = newTotal % 60;
        setManualTime((newHour % 24 + 24) % 24, (newMinute + 60) % 60);
    }

    function startDrag(e) {
        clockContainer.classList.add('dragging'); isDragging = true; stopClock();
        const rect = clockContainer.getBoundingClientRect(), cx = rect.left + rect.width / 2, cy = rect.top + rect.height / 2;
        const clientX = e.touches ? e.touches[0].clientX : e.clientX, clientY = e.touches ? e.touches[0].clientY : e.clientY;
        const angRad = Math.atan2(clientY - cy, clientX - cx);
        lastAngle = (angRad * 180 / Math.PI + 90 + 360) % 360; cumulativeMinuteDelta = 0;
        const [hh, mm] = digitalTimeEl.textContent.split(':'); initialHour = parseInt(hh, 10) || 0; initialMinute = parseInt(mm, 10) || 0;
        handleInteraction(e);
    }
    function endDrag(){ isDragging = false; clockContainer.classList.remove('dragging'); }

    timeModeBtn.addEventListener('click', () => {
        if (isRealTime) { stopClock(); const now = new Date(); setManualTime(now.getHours(), now.getMinutes()); }
        else { startClock(); }
    });
    clockContainer.addEventListener('mousedown', startDrag);
    clockContainer.addEventListener('touchstart', startDrag, { passive: false });
    window.addEventListener('mousemove', handleInteraction, { passive: false });
    window.addEventListener('touchmove', handleInteraction, { passive: false });
    window.addEventListener('mouseup', endDrag); window.addEventListener('touchend', endDrag);

    function showFeedback(message){ if(!copyFeedback) return; copyFeedback.textContent=message; setTimeout(()=>{ copyFeedback.textContent=''; },2000); }
    if (shareBtn) shareBtn.addEventListener('click', () => {
        const shareData = { title:'Lebanese Arabic Clock', text:'Learn to tell time in Lebanese Arabic with this interactive tool!', url: location.href };
        if (navigator.share) navigator.share(shareData).catch(()=>{});
        else { navigator.clipboard.writeText(location.href).then(()=>showFeedback('Link copied to clipboard!')); }
    });
    if (embedBtn) embedBtn.addEventListener('click', () => {
        const embedCode = `<iframe src="${location.href}" title="Lebanese Arabic Clock" loading="lazy" width="100%" height="600" style="border:0; max-width:480px; display:block; margin-inline:auto;"></iframe>`;
        navigator.clipboard.writeText(embedCode).then(()=>showFeedback('Embed code copied!')).catch(()=>showFeedback('Failed to copy.'));
    });

    // --- All functions defined, now run the initialization ---
    try {
        drawClockMarkers();
        startClock();
        
        // This nested requestAnimationFrame is the fix for the mobile bug.
        requestAnimationFrame(() => {
            // Frame 1: Hide the spinner
            spinner.style.display = 'none';
            wrapper.style.minHeight = 'auto';
            
            // Frame 2: Start the fade-in
            requestAnimationFrame(() => {
                clockApp.classList.add('loaded');
            });
        });

    } catch (error) {
        console.error("Failed to initialize clock:", error);
        if(spinner) spinner.style.display = 'none';
        if(wrapper) wrapper.innerHTML = '<p style="text-align:center; color:red;">Sorry, the clock could not be loaded.</p>';
    }
  }

  // ---- FINAL ROBUST INITIALIZATION LOGIC V4 ----
  const elementCheckInterval = setInterval(function() {
    if (document.getElementById('lebanese-clock-app-wrapper')) {
      clearInterval(elementCheckInterval);
      init();
    }
  }, 50);

  // Failsafe timeout
  setTimeout(() => {
    init(); // init() function has a check to not run more than once.
  }, 4000);

})();
</script>
